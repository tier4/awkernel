# Automatically load overlays for detected cameras

otg_mode=1

[pi3]

[pi4]
#enable_jtag_gpio=1
enable_gic=1

# Run as fast as firmware / board allows
arm_boost=1

[all]
# boot AARCH64 mode
arm_64bit=1

# kernel file
kernel=kernel8.img

disable_commandline_tags=1

# set uart clock to 48MHz
init_uart_clock=48000000
